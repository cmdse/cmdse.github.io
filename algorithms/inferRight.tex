\documentclass[varwidth=100cm]{standalone}
\usepackage[utf8]{inputenc}
\usepackage{csquotes}
\usepackage{algpseudocode}

\begin{document}
\begin{algorithmic}[1]
  \Require{$token.type$ is not \enquote{semantic}}
  \Statex
  \Function{inferRight}{$token$, $tokenList$}
    \State{$index \gets token.argumentPosition$}
    \If{$token.type$ is \enquote{option part} \textbf{and} $index < tokenList.length + 1$}
      \State{$neighbour \gets tokenList[index+1]$}
      \If{$neighbour.type.posModel.binding$ is (\enquote{NONE} or \enquote{RIGHT})}
          \State{\textbf{remove} $type$ \textbf{from} $token.semanticCandidates$ \textbf{when} $type.posModel.binding$ is \enquote{RIGHT}}
          \State{$convertToSemantic(token)$}
      \EndIf
    \EndIf
  \EndFunction
\end{algorithmic}
\end{document}
